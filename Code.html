                        function getValues() {
                        
                                // get the text input from the page
                                let message = document.getElementById('userInput').value;
                                // validate the input: make sure the input no empty
                        
                                if (message.length == 0) {
                                    Swal.fire ({
                                        icon: 'error',
                                        backdrop: false,
                                        title: 'Oops!',
                                        text: 'Please enter a string to reverse'
                                });
                                } else {
                                // pass the input to reverseMessage function and assign its return value to a variable
                                let revMsg = reverseMessage(message);
                                // give the reversed message to displayMessage
                                displayMessage(revMsg);
                        
                                }                        
                        }                       
                        
                        function reverseMessage(input) {

                            let output = '';
                            for (let i = input.length - 1; i >= 0; i--) {
                                let letter = input[i];
                                output += letter;
                            }
                            return output;
                        }
                        
                        function displayMessage(message) {
                            document.getElementById('msg').textContent = `Your message reversed is: ${message}`;
                            document.getElementById('alert').classList.remove('invisible');
                        }

The code is in three different section:

The first section:

getValues() collect the input data that the user provide.

The second section:

reverseMessage(input) use the value from getValues and loop through each letter to get the input in backward order.

The last section:

displayMessage(message) display the result from reverseMessage to the end user.
